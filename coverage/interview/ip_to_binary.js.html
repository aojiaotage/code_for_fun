function validateNumberStr(nStr) {
  const n = Number(nStr);
  if (Number.isNaN(n)) throw new Error(`str: "${nStr}" is not a number!`);
  if (n < 0 || n > 255) throw new Error(`str: "${nStr}" is out of range 0-255`);
  return true;
}
